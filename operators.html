11. Enter marks of students and calculate average of students marks ?

show code 
    #include < stdio.h >

    int main() {
    int n;
    printf("Enter number of subjects: ");
    scanf("%d", &n);

    float marks, sum = 0;
    for (int i = 1; i <= n; i++) {
        printf("Enter marks of subject %d: ", i);
        scanf("%f", &marks);
        sum += marks;
    }

    float average = sum / n;
    printf("Average marks = %.2f\n", average);
    return 0;
   }
show output

Output:

Enter number of subjects: 3 Enter marks of subject 1: 75 Enter marks of subject 2: 80 Enter marks of subject 3: 85 Average marks = 80.00

12. Calculate simple interest = (P x R x T)/ 100?

show code 
   #include < stdio.h >

    int main() {
    float P, R, T, SI;
    printf("Enter Principal amount: ");
    scanf("%f", &P);
    printf("Enter Rate of interest: ");
    scanf("%f", &R);
    printf("Enter Time (in years): ");
    scanf("%f", &T);

    SI = (P * R * T) / 100;
    printf("Simple Interest = %.2f\n", SI);
    return 0;
    }
show output

Output:

Enter Principal amount: 1000 Enter Rate of interest: 5 Enter Time (in years): 2 Simple Interest = 100.00
